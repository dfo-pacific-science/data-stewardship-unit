// Calculate executive summary metrics
executiveMetrics = {
  const rowsWithScores = rawRows.map(row => ({
    ...row,
    score: parseFloat(row.Score || 0)
  })).filter(row => row.score > 0);

  // Calculate average score per Level 2 result
  const level2ScoreData = new Map();
  rowsWithScores.forEach(row => {
    const l2 = row[COL_L2];
    if (l2) {
      if (!level2ScoreData.has(l2)) {
        level2ScoreData.set(l2, { scores: [], name: l2 });
      }
      level2ScoreData.get(l2).scores.push(row.score);
    }
  });

  // Calculate average for each Level 2 result
  const level2Averages = Array.from(level2ScoreData.entries())
    .map(([l2, data]) => ({
      name: l2,
      avgScore: data.scores.reduce((sum, s) => sum + s, 0) / data.scores.length,
      count: data.scores.length
    }))
    .sort((a, b) => b.avgScore - a.avgScore)
    .slice(0, 5);

  return {
    top5Results: level2Averages
  };
}

// Calculate Level 2 results data for horizontal stacked bar chart
level2StackedData = {
  // Normalize Level 2 values to handle whitespace variations and common typos
  function normalizeLvl2(text) {
    if (!text) return text;
    // Fix missing space after possessive apostrophes (e.g., "Canada'soceans" -> "Canada's oceans")
    let normalized = text.replace(/'s([a-z])/gi, "'s $1");
    // Trim and normalize multiple spaces to single space
    normalized = normalized.trim().replace(/\s+/g, ' ');
    // Normalize "and other" to "and" to combine similar results
    normalized = normalized.replace(/\s+and\s+other\s+/gi, ' and ');
    // Final whitespace normalization
    normalized = normalized.replace(/\s+/g, ' ').trim();
    return normalized;
  }

  const rowsWithScores = rawRows.map(row => ({
    ...row,
    score: parseFloat(row.Score || 0) || 0
  })).filter(row => row.score > 0 && row[COL_L2] && row.Program);

  // Normalize Level 2 values first to handle whitespace variations
  const normalizedRows = rowsWithScores.map(row => ({
    ...row,
    normalizedL2: normalizeLvl2(row[COL_L2])
  }));

  // Group by normalized Level 2 and Program, collecting ALL scores from all Level 3 items
  // This rolls up all Level 3 results/indicators/plans under their Level 2 grouping
  const level2ProgramData = new Map();
  normalizedRows.forEach(row => {
    const l2 = row.normalizedL2;
    const program = row.Program;
    const key = `${l2}|${program}`;

    if (!level2ProgramData.has(key)) {
      level2ProgramData.set(key, {
        level2: l2,
        originalL2: row[COL_L2], // Keep original for display if needed
        program: program,
        scores: []
      });
    }
    // Collect all scores from all Level 3 items for this Level 2 + Program combination
    level2ProgramData.get(key).scores.push(row.score);
  });

  // Calculate average score for each Level 2 + Program combination
  level2ProgramData.forEach((data, key) => {
    const sum = data.scores.reduce((acc, s) => acc + s, 0);
    data.score = sum / data.scores.length;
    delete data.scores; // Clean up, no longer needed
  });

  // Calculate total (sum of averages) per Level 2 for sorting
  const level2Totals = new Map();
  level2ProgramData.forEach((data, key) => {
    const total = level2Totals.get(data.level2) || 0;
    level2Totals.set(data.level2, total + data.score);
  });

  // Helper function to truncate long labels
  function truncateLabel(label, maxLength = 40) {
    if (!label || label.length <= maxLength) return label;
    // Try to truncate at a word boundary if possible
    const truncated = label.substring(0, maxLength);
    const lastSpace = truncated.lastIndexOf(' ');
    if (lastSpace > maxLength * 0.7) {
      return truncated.substring(0, lastSpace) + '...';
    }
    return truncated + '...';
  }

  // Create a map from original to truncated labels, ensuring uniqueness
  const labelMap = new Map();
  const allLevel2s = Array.from(new Set(Array.from(level2ProgramData.values()).map(d => d.level2)));
  const usedLabels = new Set();

  allLevel2s.forEach(l2 => {
    let truncated = truncateLabel(l2);
    let uniqueLabel = truncated;
    let counter = 1;

    // Ensure uniqueness by appending a number if needed
    while (usedLabels.has(uniqueLabel)) {
      const suffix = ` (${counter})`;
      const maxLen = 40 - suffix.length;
      const base = truncateLabel(l2, maxLen);
      uniqueLabel = base + suffix;
      counter++;
    }

    usedLabels.add(uniqueLabel);
    labelMap.set(l2, uniqueLabel);
  });

  // Convert to array and apply truncated labels, ensuring all required fields exist
  const dataArray = Array.from(level2ProgramData.values())
    .filter(d => d && d.level2 && d.program && d.score !== undefined)
    .map(d => ({
      level2: labelMap.get(d.level2) || d.level2 || "",
      program: d.program || "",
      score: d.score || 0
    }));

  // Create sorted list with truncated labels, filtering out any undefined values
  const sortedLevel2s = Array.from(level2Totals.entries())
    .filter(([level2]) => level2)
    .sort((a, b) => b[1] - a[1])
    .map(([level2]) => labelMap.get(level2) || level2)
    .filter(l2 => l2);

  // Create a map for ordering
  const level2Order = new Map(sortedLevel2s.map((l2, i) => [l2, i]));

  // Sort data: first by Level 2 order, then by program name for consistent stacking
  dataArray.sort((a, b) => {
    const orderDiff = (level2Order.get(a.level2) || 999) - (level2Order.get(b.level2) || 999);
    if (orderDiff !== 0) return orderDiff;
    return a.program.localeCompare(b.program);
  });

  return {
    data: dataArray,
    level2Order: sortedLevel2s
  };
}

prepareGraph = (rows) => {
  const nodes = new Map();
  const edges = new Map();
  const children = new Map();
  const evidence = new Map();
  const touches = new Map();

  function ensureNode(type, label) {
    if (!label) return null;
    const id = `${type}:${slugify(label)}`;
    if (!nodes.has(id)) {
      const color = type === "Program" && programColors[label] ? programColors[label] : typeMeta[type].color;
      nodes.set(id, { id, label, type, level: levelByType[type], color });
    }
    return nodes.get(id);
  }

  function connect(source, target) {
    if (!source || !target) return;
    const id = `${source.id}->${target.id}`;
    if (!edges.has(id)) {
      const label = source.type === "Program" ? source.label : "";
      edges.set(id, { id, source: source.id, target: target.id, label });
    }
  }

  function addTouch(node) {
    if (!node) return;
    touches.set(node.id, (touches.get(node.id) ?? 0) + 1);
  }

  for (const row of rows) {
    const level1 = ensureNode("Level1", row[COL_L1]);
    const level2 = ensureNode("Level2", row[COL_L2]);
    const level3 = ensureNode("Level3", row[COL_L3]);
    const program = ensureNode("Program", row.Program);

    connect(level1, level2);
    connect(level2, level3);
    connect(program, level3);

    if (row.Program) {
      addTouch(level1);
      addTouch(level2);
      addTouch(level3);
      addTouch(program);
    }

    if (level3) {
      const bucket = evidence.get(level3.id) ?? { programs: new Set(), snippets: new Set() };
      if (row.Program) bucket.programs.add(row.Program);
      if (row["Program Evidence"]) bucket.snippets.add(row["Program Evidence"]);
      evidence.set(level3.id, bucket);
    }
  }

  for (const edge of edges.values()) {
    if (!children.has(edge.source)) children.set(edge.source, new Set());
    children.get(edge.source).add(edge.target);
  }

  const nodesList = [...nodes.values()];
  const edgesList = [...edges.values()];
  for (const n of nodesList) {
    n.programTouches = touches.get(n.id) ?? 0;
  }
  const initialVisible = nodesList.filter((n) => n.level <= 1).map((n) => n.id);
  const roots = nodesList.filter((n) => n.level === 0).map((n) => n.id);
  const nodeById = new Map(nodesList.map((n) => [n.id, n]));
  const evidenceMap = new Map(
    [...evidence.entries()].map(([id, data]) => [
      id,
      {
        programs: [...data.programs].sort(),
        snippets: [...data.snippets].sort()
      }
    ])
  );
  const childrenMap = new Map([...children.entries()].map(([id, set]) => [id, [...set]]));

  return { nodes: nodesList, nodeById, edges: edgesList, roots, initialVisible, evidence: evidenceMap, children: childrenMap };
};

        padding: "6px",
        "border-color": "#0e2437",
        "border-width": 1.25,
        width: 30,
        height: 30
      }
    },
    {
      selector:
        "[type = 'Level2'], [type = 'Level3']",
      style: {
        width: "mapData(programTouches, 0, 25, 32, 90)",
        height: "mapData(programTouches, 0, 25, 32, 90)"
      }
    },
    {
      selector: "edge",
      style: {
        width: 1.6,
        "curve-style": "bezier",
        "line-color": "#b4c7d6",
        "target-arrow-color": "#b4c7d6",
        "target-arrow-shape": "triangle",
        label: "data(label)",
        "font-size": 9,
        "text-background-color": "#ffffff",
        "text-background-opacity": 0.75,
        "text-background-padding": 2,
        "text-rotation": "autorotate",
        "text-margin-x": 2,
        "text-margin-y": -2
      }
    },
    {
      selector: "node:selected",
      style: {
        "border-color": "#d9731e",
        "border-width": 3
      }
    }
  ];

  const layoutOptions = () => ({
    name: "breadthfirst",
    directed: true,
    padding: 30,
    spacingFactor: 1.15
  });

  const elementsFromVisible = () => {
    const nodes = graphData.nodes
      .filter((n) => visible.has(n.id))
      .map((n) => ({ data: { ...n, typeLabel: typeMeta[n.type].label } }));
    const edges = graphData.edges
      .filter((e) => visible.has(e.source) && visible.has(e.target))
      .map((e) => ({ data: e }));
    return [...nodes, ...edges];
  };

  function runLayout() {
    if (!cy || !cy.nodes().length) return;
    try {
      cy.layout(layoutOptions()).run();
    } catch (error) {
      cy.layout({ name: "grid", padding: 30 }).run();
    }
  }

  function refresh(runLayoutFlag = true) {
    if (!cy) return;
    cy.elements().remove();
    cy.add(elementsFromVisible());
    if (runLayoutFlag) runLayout();
    updateCounts();
  }

  function updateCounts() {
    const edgeCount = graphData.edges.filter((e) => visible.has(e.source) && visible.has(e.target)).length;
    counts.textContent = `${visible.size} nodes / ${edgeCount} edges shown`;
  }

  function addChildren(nodeId) {
    const kids = graphData.children.get(nodeId) || [];
    for (const kid of kids) visible.add(kid);
  }

  function addChildrenForVisible() {
    for (const nodeId of [...visible]) addChildren(nodeId);
  }

  function resetView() {
    visible = new Set(graphData.initialVisible);
    refresh(true);
    renderDetails(null);
  }

  function renderDetails(nodeId) {
    detail.innerHTML = "";
    if (!nodeId) {
      detail.appendChild(html`<div class="meta">Click a node to see its details.</div>`);
      return;
    }
    const node = graphData.nodeById.get(nodeId);
    if (!node) return;

    const header = html`<div>
      <h4>${node?.label || ""}</h4>
      <div class="meta">${typeMeta[node?.type]?.label || ""}</div>
    </div>`;

    const childList = graphData.children.get(nodeId) || [];
    const childInfo =
      childList.length > 0
        ? html`<div class="meta">${childList.length} immediate child${childList.length === 1 ? "" : "ren"}.</div>`
        : html`<div class="meta">No children in the current filtered data.</div>`;

    detail.append(header, childInfo);

    const evidence = graphData.evidence.get(nodeId);
    if (evidence) {
      if (evidence.programs.length) {
        detail.appendChild(
          html`<div class="meta"><strong>Contributing programs:</strong> ${evidence?.programs?.join(", ") || "None"}</div>`
        );
      }
      if (evidence.snippets.length) {
        detail.appendChild(html`<div class="results-graph-evidence-title">Evidence</div>`);
        const list = html`<ul>${(evidence?.snippets || []).filter(s => s).map((s) => html`<li>${s || ""}</li>`)}</ul>`;
        detail.appendChild(list);
      }
    }
  }

  resetBtn.onclick = () => resetView();
  expandBtn.onclick = () => {
    addChildrenForVisible();
    refresh(true);
  };

  function initGraph() {
    if (!canvas.isConnected) {
      requestAnimationFrame(initGraph);
      return;
    }

    cy = cytoscape({
      container: canvas,
      elements: elementsFromVisible(),
      style: baseStyle,
      wheelSensitivity: 0.2
    });

    runLayout();

    cy.on("tap", "node", (evt) => {
      const id = evt.target.id();
      addChildren(id);
      refresh(true);
      renderDetails(id);
    });

    updateCounts();
  }

  requestAnimationFrame(initGraph);
  invalidation.then(() => {
    if (cy) cy.destroy();
  });

  return shell;
}
